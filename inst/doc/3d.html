has_X11 = Sys.getenv("DISPLAY") != ""
if (has_X11) knit_hooks$set(webgl = hook_webgl)

point 3d graph

x <- sort(rnorm(1000))
y <- rnorm(1000)
z <- rnorm(1000) + atan2(x, y)
plot3d(x, y, z, col = rainbow(1000))
